    ```typescript
    export class ExpertAPI {
      constructor(
        private skillRegistry: SkillRegistry,
        private agentBuilder: AgentBuilder
      ) {}

      async invokeExpert(
        expertId: string,
        task: ExpertTaskInput
      ): Promise<ExpertOutput> {
        const startTime = Date.now();
        const expert = this.getExpertDefinition(expertId);
        const skill = expert.createSkill();

        // Validate required skills are available in registry
        for (const skillName of expert.skills) {
          if (!this.skillRegistry.getMetadata(skillName)) {
            throw new Error(`Skill ${skillName} not found in registry`);
          }
        }

        try {
          const result = await skill.execute(task);
          return {
            ...result,
            json: {
              ...result.json,
              metadata: {
                ...result.json.metadata,
                durationMs: Date.now() - startTime
              }
            }
          };
        } catch (error) {
          // Return partial results with error per CONTEXT.md
          return this.formatErrorOutput(error, task);
        }
      }
    }
    ```
